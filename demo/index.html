Example 4: Input hidden based

Javascript


$(':input[name=hiddenimg]').inputFileThumb({
	'title'                 : 'Picture by ID',
	'extensions'            : ['png','jpg','jpeg','gif'],
	choose : function() {
		var file = prompt('The file ?');
		if (file) {
			var self = $(this);
			self.inputFileThumb('option', 'file', file);
			if (self.inputFileThumb('option', 'file') === file) {
				self.val(file);
			}
		}
	}
});

Example 5: Not :input based

Javascript


$('#divimg').inputFileThumb({
	'title'                 : 'Choisissez un document',
	'extensions'            : ['pdf', 'doc', 'odt'],
	'extensionsType'        : { // Predefined metatypes of files with associated extensions
	      'document'   : ['pdf', 'doc', 'odt']
    },
	'texts'                 : {
	   'add'            : 'Ajouter',
	   'edit'           : 'Modifier',
	   'delete'         : 'Supprimer',
	   'wrongExtension' : 'Cette extension  n\'est pas autorisée'
	},
	choose : function() {
		var file = prompt('The file ?');
		if (file) {
			var self = $(this);
			self.inputFileThumb('option', 'file', file);
			if (self.inputFileThumb('option', 'file') === file) {
				alert(file);
			}
		}
	}
});
