Example 1: the default InputFileThumb

Javascript


$(':file[name=simple]').inputFileThumb({    
    'title'                 : 'The file title',
    'description'           : 'Here more info about the file'
});

Example 2: A PNG is required

Javascript


$(':file[name=pngonly]').inputFileThumb({    
	'title'                 : 'PNG image',
	'description'           : 'Please select a PNG file',
	'extensions'            : ['png']
});

Example

Example 3: An image has been uploaded

Javascript


$(':file[name=existingimg]').inputFileThumb({    
	'file'                  : 'vignette.png', 
	'title'                 : 'Picture', 
	'description'           : 'Pick a nice shot!',
	'extensions'            : ['png','jpg','jpeg','gif'], 
	'deleteInput'           : 'delete_hidden'
});
